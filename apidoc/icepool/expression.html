290        self,
291        op_name: SetComparatorStr,
292        other: 'MultisetExpression | Mapping[T, int] | Collection[T]',
293        /,
294        truth_value: bool | None = None
295    ) -> 'icepool.MultisetEvaluator[T, int, bool]':
296        """Compares the outcome multiset to another multiset.
297
298        You can also use the symbolic operators directly, e.g.
299        `generator <= [1, 2, 2]`.
300
301        If the other argument is a `Mapping` or `Collection`, it
302        will be treated as a fixed target.
303
304        Args:

306                `<, <=, >, >=, ==, !=`.
307            other: The right-side generator or multiset to compare with.
308        """
309        if isinstance(other, MultisetExpression):
310            evaluator = icepool.evaluator.ComparisonEvaluator.new_by_name(
311                op_name)  # type: ignore
312            return icepool.expression.ExpressionEvaluator(
313                self, other, evaluator=evaluator, truth_value=truth_value)
314        elif isinstance(other, (Mapping, Collection)):
315            evaluator = icepool.evaluator.ComparisonEvaluator.new_by_name(
316                op_name, other)
317            return icepool.expression.ExpressionEvaluator(
318                self, evaluator=evaluator, truth_value=truth_value)
319        else:
320            return NotImplemented
